#include "HX711.h"
#include <Wire.h> 
#include <LiquidCrystal_I2C.h>

HX711 scale(6, 5);
LiquidCrystal_I2C lcd(0x3F, 16, 2);

float calibration_factor = -95.00;
float units;
float ounces;
int min_value=280;

void send_sms()
{
  
  Serial.begin(9600);   
  Serial.begin(9600);     
  delay(5000);
  Serial.println("AT+CMGF=1");
  delay(1000);
  Serial.println("AT+CMGS=\"+91**********\"\r");
  delay(1000);
  Serial.println("LOW GAS");
  delay(100);
  Serial.println((char)26);
  delay(1000);
}

void buzzor()
{

      digitalWrite(3,HIGH);
      delay(5000);
      digitalWrite(3,LOW);

}

void setup()
{
  pinMode(11,OUTPUT);//display
  pinMode(3,OUTPUT);//buzzer
  digitalWrite(11,HIGH);
  Serial.begin(9600);
  Serial.println("HX711 weighing");
  scale.set_scale(calibration_factor);
  scale.tare();
  Serial.println("Readings:");
  lcd.begin();
  lcd.backlight();
  lcd.clear();
}

void loop()
{
  delay(3000);
  Serial.print("Reading:");
  units = scale.get_units(),10;
  if (units < 0)
  {
    units = 0.00;
  }
  ounces = units * 0.035274;
  if(units<100)
  {
    lcd.setCursor(0,0);
    lcd.print("  Kindly Place");
    lcd.setCursor(0,1);
    lcd.print("  the Cylinder");

  }
  else
  {
  		lcd.clear();
  		Serial.print(units);
  		lcd.setCursor(0,0);
  		lcd.print("Weight[kg]");
  		lcd.setCursor(0,1);
  		lcd.print(units/1000);
  		Serial.println(" grams");
  		delay(1000);
  		if(units<min_value)
  		{
      	send_sms();
      	buzzor();
      
  		}
	}
}
